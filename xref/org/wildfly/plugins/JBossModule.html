64      public static JBossModule readFromZipFile(Log log, File moduleZip) throws Exception {
65          JBossModule m = new JBossModule(log);
66          m.isZip = true;
67          m.root = moduleZip;
68          if (!(moduleZip.canRead() && moduleZip.isFile())) {
69              throw new FileNotFoundException("File " + moduleZip.getAbsolutePath() + " does not exist");
70          }
71          ZipInputStream zin = null;
72          BufferedInputStream bin = null;
73          boolean moduleXmlFound = false;
74          try {
75              bin = new BufferedInputStream(new FileInputStream(moduleZip));
76              zin = new ZipInputStream(bin);
77              ZipEntry ze = null;
78              while ((ze = zin.getNextEntry()) != null) {
79                  if (ze.getName().endsWith("main/module.xml")) {
80                      moduleXmlFound = true;
81                      DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
82                      DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
83                      Document doc = dBuilder.parse(zin);
84                      m.readModuleXmlInfo(doc);
85                      break;
86                  }
87              }
88          } finally {
89              try {
90                  // it should be enough to close the latest created stream in
91                  // case of chained (nested) streams, @see
92                  // http://www.javapractices.com/topic/TopicAction.do?Id=8
93                  if (zin != null) {
94                      zin.close();
95                  }
96              } catch (IOException ex) {
97              }
98          }
99          if (!moduleXmlFound) {
100             throw new FileNotFoundException("module.xml was not found in " + moduleZip.getAbsolutePath());
101         }
102         return m;
103     }

105     private void readModuleXmlInfo(Document doc) throws Exception {
106         this.moduleId = doc.getDocumentElement().getAttribute("name");
107         XPath xPath = XPathFactory.newInstance().newXPath();
108         NodeList nodeList = (NodeList) xPath.compile("//resources/resource-root").evaluate(doc, XPathConstants.NODESET);
109         for (int i = 0; i < nodeList.getLength(); i++) {
110             this.resources.add(nodeList.item(i).getAttributes().getNamedItem("path").getTextContent());
111         }
112     }

114     public static JBossModule readFromDir(Log log, File rootDir) throws Exception {
115         JBossModule m = new JBossModule(log);
116         m.root = rootDir;
117         File moduleFile = new File(rootDir, "module.xml");
118         if (!moduleFile.exists() || !moduleFile.canRead()) {
119             throw new FileNotFoundException("File " + moduleFile.getAbsolutePath() + " does not exist");
120         }
121         DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
122         DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
123         Document doc = dBuilder.parse(moduleFile);
124         m.readModuleXmlInfo(doc);
125         return m;
126     }

128     /**
129      * installs JBossModule to given directory
130      * @param jbossHome target directory to install module
131      * @return list of installed files
132      * @throws Exception
133      */
134     public List<File> installTo(File jbossHome) throws Exception {
135         List<File> installedFiles = new ArrayList<File>();
136         if (isZip) {
137             ZipInputStream zin = null;
138             FileOutputStream fos;
139             log.info("Extracting module [" + this.root.getAbsolutePath() + "] to [" + jbossHome.getAbsolutePath() + "]");
140             try {
141                 zin = new ZipInputStream(new FileInputStream(this.root));
142                 ZipEntry ze = null;
143                 while ((ze = zin.getNextEntry()) != null) {
144                     String fileName = ze.getName();
145                     File newFile = new File(jbossHome + File.separator + fileName);
146                     if (!ze.isDirectory()) {
147                         log.debug("Writing " + newFile.getAbsolutePath());
148                         File parent = newFile.getParentFile();
149                         if (!parent.exists()) {
150                             parent.mkdirs();
151                         }
152                         fos = new FileOutputStream(newFile);
153                         IOUtil.copy(zin, fos);
154                         IOUtil.close(fos);
155                         installedFiles.add(newFile);
156                     }
157                 }
158             } finally {
159                 try {
160                     // it should be enough to close the latest created stream in
161                     // case of chained (nested) streams, @see
162                     // http://www.javapractices.com/topic/TopicAction.do?Id=8
163                     if (zin != null) {
164                         zin.close();
165                     }
166                 } catch (IOException ex) {
167                 }
168             }
169         } else {
170             File targetDir = new File(jbossHome, "modules" + File.separator + moduleId.replaceAll("\\.", File.separator) + File.separator + "main");
171             if (!targetDir.exists() && !targetDir.mkdirs()) {
172                 throw new MojoFailureException("Failed to create module directory " + targetDir.getAbsolutePath());
173             }
174             File moduleFile = new File(this.root, "module.xml");
175             for (String r : resources) {
176                 File resource = new File(root, r);
177                 if (!(resource.canRead())) {
178                     throw new MojoFailureException("Resource file [" + r + "] referenced in [" + moduleFile.getAbsolutePath()
179                             + "] does not exist, fix ");
180                 }
181                 if (resource.isFile()) {
182                     FileUtils.copyFileToDirectory(resource, targetDir);
183                     installedFiles.add(resource);
184                 } else if (resource.isDirectory()) {
185                     FileUtils.copyDirectoryStructure(resource, targetDir);
186                 }
187             }
188             FileUtils.copyFileToDirectory(moduleFile, targetDir);
189         }
190         return installedFiles;
191     }
