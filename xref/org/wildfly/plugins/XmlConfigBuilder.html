111     public void build() throws Exception {
112         DocumentBuilder dBuilder = factory.newDocumentBuilder();
113         Document srcDoc = dBuilder.parse(sourceFile);
114 
115         // sort inserts by the shortest
116         Collections.sort(getInserts(), new Comparator<Insert>() {
117 
118             public int compare(Insert o1, Insert o2) {
119                 Integer o1Size = o1.getSelect().split("/").length;
120                 Integer o2Size = o2.getSelect().split("/").length;
121                 return o1Size.compareTo(o2Size);
122             }
123         });
124         debug("Building [" + this.sourceFile + "] ");
125         // if our target document defines namespace in root element, let's read
126         // it
127         String namespace = getNameSpace(srcDoc);
128         if (namespace != null) {
129             namespaceContext.mapping(PREFIX, namespace);
130         }
131 
132         for (Insert insert : getInserts()) {
133             debug("Applying " + insert);
134             String expression = insert.getSelect();
135 
136             if (namespace != null) {
137                 // enhance given xpath to use namespaces
138                 expression = xpath2Namespaced(expression, PREFIX);
139                 debug("Expression " + expression);
140             }
141             XPathExpression expr = xpath.compile(expression);
142             try {
143                 NodeList nl = (NodeList) expr.evaluate(srcDoc, XPathConstants.NODESET);
144                 if (nl.getLength() == 0) {
145                     if (failNoMatch) {
146                         throw new Exception("Failed to update [" + targetFile.getAbsolutePath() + "] " + insert + " does not select any element");
147                     }
148                     warning(insert + " does not select any element");
149                     continue;
150                 }
151                 debug("Expression evaluated to " + nl.getLength() + " nodes");
152                 Document contentDoc = null;
153                 if (insert.getContent() != null) {
154                     debug("Loading content XML from file " + insert.getContent());
155                     contentDoc = dBuilder.parse(insert.getContent());
156                 } else {
157                     debug("Loading content XML from string");
158                     contentDoc = dBuilder.parse(new ByteArrayInputStream(insert.getXml().getBytes()));
159                 }
160 
161                 for (int i = 0; i < nl.getLength(); i++) {
162                     Node node = nl.item(i);
163                     if (node instanceof Element) {
164                         Element element = (Element) node;
165                         Node inserting = contentDoc.getDocumentElement().cloneNode(true);
166                         srcDoc.adoptNode(inserting);
167                         String recentNs = findRecentNamespaceFromXpath(expression);
168                         // is the root node of inserting content already
169                         // present?
170 
171                         XPathExpression contentExpr = createContentRootExpression(contentDoc, recentNs, namespace, insert.getAttribute());
172                         NodeList existingNodes = (NodeList) contentExpr.evaluate(element, XPathConstants.NODESET);
173                         if (existingNodes.getLength() > 0) {
174                             // we need to remove those? (could be many)
175                             // we'll replace the last guy
176                             element.replaceChild(inserting, existingNodes.item(existingNodes.getLength() - 1));
177                         } else {
178                             element.appendChild(inserting);
179                         }
180 
181                         String contentNs = getNameSpace(contentDoc);
182                         // find most recent NS from inserted node back to root
183                         // node and assign to it
184                         recentNs = findRecentNamespace(srcDoc, inserting);
185                         if (contentNs == null && recentNs != null) {
186                             // content document does not have namespace, let's
187                             // rename it to our namespace
188                             renameNamespaceRecursive(srcDoc, inserting, recentNs);
189                         }
190                     }
191                 }
192             } catch (Exception e) {
193                 e.printStackTrace();
194             }
195         }
196         writeTargetDcoument(srcDoc);
197     }

199     /**
200      * looks for most recent namespace query in xpath (denoted by namespace-uri()='')
201      * 
202      * @param expression
203      * @return
204      */
205     public static String findRecentNamespaceFromXpath(String expression) {
206         Matcher m = NS_IN_XPATH.matcher(expression);
207         String ns = null;
208         while (m.find()) {
209             ns = m.group(1);
210         }
211         return ns;
212     }

214     private String findRecentNamespace(Document doc, Node node) {
215         Node parent = null;
216         while ((parent = node.getParentNode()) != null) {
217             Element parentEl = (Element) parent;
218             String ns = parentEl.getAttribute("xmlns");
219             if (!ns.isEmpty()) {
220                 return ns;
221             }
222             if (parent.isSameNode(doc.getDocumentElement())) {
223                 return null;
224             }
225             if (parent.isEqualNode(node)) {
226                 return null;
227             }
228             node = parent;
229         }
230         return null;
231     }

233     private void writeTargetDcoument(Document doc) throws Exception {
234         debug("Writing target file..");
235         TransformerFactory transformerFactory = TransformerFactory.newInstance();
236         Transformer transformer = transformerFactory.newTransformer();
237         transformer.setOutputProperty(OutputKeys.INDENT, "yes");
238         transformer.setOutputProperty(OutputKeys.METHOD, "xml");
239         DOMSource source = new DOMSource(doc);
240         StreamResult result = new StreamResult(targetFile);
241 
242         transformer.transform(source, result);
243         debug("Rewriting (formatting) target file..");
244         formatXmlDocument(this.targetFile);
245     }

247     public static String xpath2Namespaced(String expression, String prefix) {
248         String[] components = expression.split("/");
249         StringBuilder sb = new StringBuilder();
250         for (int i = 0; i < components.length; i++) {
251             String piece = components[i];
252             if (piece.matches("^\\w+.*")) {
253                 piece = prefix + ":" + piece;
254             }
255             sb.append(piece + "/");
256         }
257         if (sb.length() > 0) {
258             sb.deleteCharAt(sb.length() - 1);
259         }
260         return sb.toString();
261     }

267     public static String element2Xpath(Element element, String prefix, String defaultPrefix, String identityAttribute) {
268         StringBuilder sb = new StringBuilder("/");
269         String ns = element.getAttribute("xmlns");
270         String pref = "";
271         if (!ns.isEmpty()) {
272             pref = prefix + ":";
273         } else if (defaultPrefix != null) {
274             pref = defaultPrefix + ":";
275         }
276         sb.append(pref + element.getLocalName());
277         if (identityAttribute != null && !identityAttribute.isEmpty()) {
278             sb.append("[@" + identityAttribute + "='" + element.getAttribute(identityAttribute) + "']");
279         } else {
280             // use all attributes
281             NamedNodeMap attributes = element.getAttributes();
282             if (attributes.getLength() > 0) {
283                 StringBuilder sbAttributes = new StringBuilder();
284                 sbAttributes.append("[");
285                 int validAttributes = 0;
286                 for (int i = 0; i < attributes.getLength(); i++) {
287                     Node node = attributes.item(i);
288                     if ("xmlns".equals(node.getNodeName())) {
289                         // ignore this guy as it's handled with prefixing
290                         continue;
291                     }
292                     validAttributes++;
293                     sbAttributes.append("@" + node.getNodeName() + "='" + node.getNodeValue() + "' and ");
294                 }
295                 if (validAttributes > 0) {
296                     sbAttributes.delete(sbAttributes.length() - 5, sbAttributes.length()); // delete
297                                                                                            // last
298                                                                                            // '
299                                                                                            // and
300                                                                                            // '
301                     sbAttributes.append("]");
302                     sb.append(sbAttributes.toString());
303                 }
304             }
305         }
306 
307         return sb.toString();
308     }

310     private XPathExpression createContentRootExpression(Document contentDoc, String contentNamespace, String rootNamespace, String identityAttribute)
311             throws Exception {
312         String expression = null;
313 
314         if (contentNamespace != null) {
315             expression = element2Xpath(contentDoc.getDocumentElement(), PREFIX_CONTENT, PREFIX_CONTENT, identityAttribute);
316             namespaceContext.mapping(PREFIX_CONTENT, contentNamespace);
317         } else {
318             String ns = getNameSpace(contentDoc);
319             if (ns != null) {
320                 expression = element2Xpath(contentDoc.getDocumentElement(), PREFIX_CONTENT, null, identityAttribute);
321                 namespaceContext.mapping(PREFIX_CONTENT, ns);
322             } else {
323                 expression = element2Xpath(contentDoc.getDocumentElement(), PREFIX_CONTENT, rootNamespace == null ? null : PREFIX, identityAttribute);
324             }
325         }
326 
327         debug("Content expression " + expression);
328         // our expression always starts with /, but we'll be evaluating it in
329         // context of some other node, thus it needs to be relative
330         return xpath.compile(expression.substring(1));
331     }

333     private void formatXmlDocument(File file) {
334         try {

339             throw new RuntimeException("Error formatting file " + file.getAbsolutePath(), ex);

394     private void validateInsert(Insert insert) throws IllegalArgumentException, XPathExpressionException {
395         File f = insert.getContent();
396         if (f == null && insert.getXml() == null) {
397             throw new IllegalArgumentException("Either content or xml must be specified");
398         }
399         if (f != null && !(f.isFile() && f.canRead())) {
400             throw new IllegalArgumentException("File [" + f.getAbsolutePath() + "] is not a readable file");
401         }
402         String expression = insert.getSelect();
403         if (expression.length() > 1 && expression.endsWith("/")) { // remove
404                                                                    // trailing
405                                                                    // slash
406             insert.setSelect(expression.substring(0, expression.length() - 1));
407         }
408         try {
409             xpath.compile(insert.getSelect());
410         } catch (XPathExpressionException xee) {
411             throw new XPathExpressionException(insert.getSelect() + " is not a valid xpath : " + xee.getMessage());
412         }
413 
414         // TODO validate expression must eval to NODELIST
415     }

423     private static String getNameSpace(Document doc) {
424         if (doc == null) {
425             return null;
426         }

428         if (ns.isEmpty()) {
429             return null;
430         }

434     private static void renameNamespaceRecursive(Document doc, Node node, String namespace) {
435         if (node.getNodeType() == Node.ELEMENT_NODE) {
436             Element el = (Element) node;
437             if (el.getAttribute("xmlns").isEmpty()) {
438                 doc.renameNode(node, namespace, node.getNodeName());
439             } else {
440                 return; // stop renaming ns here, since all children of this
441                         // node belong to it's ns
442             }
443 
444         }
445 
446         NodeList list = node.getChildNodes();
447         for (int i = 0; i < list.getLength(); ++i) {
448             renameNamespaceRecursive(doc, list.item(i), namespace);
449         }
450     }
