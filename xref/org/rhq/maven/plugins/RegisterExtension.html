51  	public RegisterExtension(Log log) {
52  		this.log = log;

55  	/**
56  	 * registers extension to standalone.xml
57  	 * @param srcFile source file (standalone.xml)
58  	 * @param destFile output file (standalone.xml with new stuf) 
59  	 * @param subsystem subsystem file (may not need exist)
60  	 * @param moduleId ID of JBoss Module to register as an JBoss extension
61  	 * @throws Exception
62  	 */
63  	public void register(File srcFile, File destFile, File subsystem,
64  			String moduleId) throws Exception {
65  		try {
66  			log.info("Register extension module="+moduleId);
67  			DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
68  			DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
69  			
70  			// we still need to read standalone.xml and subsystem.xml to DOM to be able to detect namespaces
71  			Document srcDoc = dBuilder.parse(srcFile);
72  			Document subsystemDoc = null;
73  			
74  			if (subsystem.isFile() && subsystem.canRead()) {
75  				subsystemDoc = dBuilder.parse(subsystem);
76  				log.info("Configuring subsystem from file "+subsystem);
77  			} else {
78  				log.info("Subsystem file ["+subsystem+"] does not exist, subsystem will not be configured");
79  			}
80  			
81  			// do XSL transformation
82  			TransformerFactory f = TransformerFactory.newInstance();
83  			Transformer t = f.newTransformer(new StreamSource(createStylesheet(
84  					getNameSpace(srcDoc), moduleId, getNameSpace(subsystemDoc),
85  					doc2string(subsystemDoc))));
86  			Source s = new StreamSource(srcFile);
87  			Result r = new StreamResult(destFile);
88  			t.setURIResolver(null);
89  			t.transform(s, r);
90  			log.info("New serverConfig file written to ["+destFile.getAbsolutePath()+"]");
91  
92  		} catch (TransformerConfigurationException e) {
93  			throw new Exception(e.toString());
94  		} catch (TransformerException e) {
95  			throw new Exception(e.toString());
96  		}
97  	}

99  	private static InputStream createStylesheet(String namespace, String module,
100 			String subns, String subsystem) throws IOException {
101 		// when writing new nodes we'll ignore wring xmlns attribute
102 		String ignoreNs = "s"; // s is main namespace for server config
103 		boolean isSubsystem = subns != null && subsystem != null;
104 
105 		StringBuilder sheet = new StringBuilder("<?xml version=\"1.0\" ?>");
106 		sheet.append("<xsl:stylesheet xmlns:xsl=\"http://www.w3.org/1999/XSL/Transform\""
107 				+ " xmlns:s=\"" + namespace + "\"");
108 		
109 		if (isSubsystem) {
110 			sheet.append(" xmlns:sub=\"" + subns + "\"");
111 			ignoreNs += " sub"; // sub is namespace for subsystem we're setting up
112 		}
113 		sheet.append(" exclude-result-prefixes=\"" + ignoreNs
114 				+ "\" version=\"1.0\">");
115 		sheet.append("<xsl:output method=\"xml\" indent=\"yes\" />");
116 
117 		// append new extension
118 		sheet.append("<xsl:template match=\"s:extensions\"><xsl:copy><xsl:apply-templates select=\"*[not(@module='"
119 				+ module
120 				+ "')]\" /><extension module=\""
121 				+ module
122 				+ "\" /></xsl:copy></xsl:template>");
123 
124 		// append new subsystem
125 		if (isSubsystem) {
126 			sheet.append("<xsl:template match=\"s:profile\"><xsl:copy><xsl:apply-templates select=\"*[not(namespace-uri() ='"
127 					+ subns
128 					+ "')]\" />"
129 					+ subsystem
130 					+ "</xsl:copy></xsl:template>");
131 		}
132 
133 		// generic identity template
134 		sheet.append("<xsl:template match=\"@*|node()\"><xsl:copy><xsl:apply-templates select=\"@*|node()\" /></xsl:copy></xsl:template>");
135 		sheet.append("</xsl:stylesheet>");
136 		return new ByteArrayInputStream(IOUtil.toByteArray(sheet.toString()));
137 	}

139 	/**
140 	 * return XML namespace for root of given document
141 	 * @param doc
142 	 * @return
143 	 */
144 	private static String getNameSpace(Document doc) {
145 		if (doc == null) {
146 			return null;
147 		}
148 		return doc.getDocumentElement().getAttribute("xmlns");
149 	}
150 
151 	/**
152 	 * return content of XML document as string
153 	 * @param doc
154 	 * @return
155 	 */
156 	private static String doc2string(Document doc) {
157 		if (doc == null) {
158 			return null;
159 		}
160 		try {
161 			StringWriter sw = new StringWriter();
162 			TransformerFactory tf = TransformerFactory.newInstance();
163 			Transformer transformer = tf.newTransformer();
164 			transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION,
165 					"yes");
166 			transformer.setOutputProperty(OutputKeys.METHOD, "xml");
167 			transformer.setOutputProperty(OutputKeys.INDENT, "yes");
168 			transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
169 			transformer.transform(new DOMSource(doc), new StreamResult(sw));
170 			return sw.toString();
171 		} catch (Exception ex) {
172 			throw new RuntimeException("Error converting to String", ex);
173 		}
174 	}
