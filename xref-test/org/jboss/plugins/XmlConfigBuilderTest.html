82  		// element without ns default prefix passed

84  		// element with ns
85  		Assert.assertEquals("/x:test", XmlConfigBuilder.element2Xpath(xml("<test xmlns=\"foo\"></test>").getDocumentElement(),"x",null));
86  		// element with attributes
87  		Assert.assertEquals("/test[@attr1='val1' and @attr2='val2']", XmlConfigBuilder.element2Xpath(xml("<test attr1=\"val1\"  attr2=\"val2\"></test>").getDocumentElement(),"x",null));
88  		// element with ns and attributes
