83      private void assertXpath(String expression, Document doc, int expectedCount) throws Exception {
84          XPathExpression expr = xpath.compile(expression);
85          NodeList nl = (NodeList) expr.evaluate(doc, XPathConstants.NODESET);
86          Assert.assertEquals(expectedCount, nl.getLength());
87      }

102     @Test
103     public void testElement2Xpath() throws Exception {
104         // element without ns
105         Assert.assertEquals("/test", XmlConfigBuilder.element2Xpath(xml("<test></test>").getDocumentElement(), "x", null));
106         // element without ns default prefix passed
107         Assert.assertEquals("/y:test", XmlConfigBuilder.element2Xpath(xml("<test></test>").getDocumentElement(), "x", "y"));
108         // element with ns
109         Assert.assertEquals("/x:test", XmlConfigBuilder.element2Xpath(xml("<test xmlns=\"foo\"></test>").getDocumentElement(), "x", null));
110         // element with attributes
111         Assert.assertEquals("/test[@attr1='val1' and @attr2='val2']",
112                 XmlConfigBuilder.element2Xpath(xml("<test attr1=\"val1\"  attr2=\"val2\"></test>").getDocumentElement(), "x", null));
113         // element with ns and attributes
114         Assert.assertEquals("/x:test[@attr1='val1' and @attr2='val2']",
115                 XmlConfigBuilder.element2Xpath(xml("<test attr1=\"val1\" xmlns=\"foo\"  attr2=\"val2\"></test>").getDocumentElement(), "x", null));
